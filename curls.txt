# Mercury + 1Claw E2E curl recipes
#
# Before running: start Mercury with real 1Claw env (see .env.example), e.g.
#   export MERCURY_ONECLAW_BASE_URL=https://api.1claw.xyz
#   export MERCURY_ONECLAW_VAULT_ID=<your-vault-uuid>
#   export MERCURY_ONECLAW_API_KEY=<your-agent-api-key>
#   export MERCURY_ONECLAW_AGENT_ID=<your-agent-uuid>   # recommended for hosted 1Claw
#   uv run uvicorn mercury.service.api:app --reload --host 127.0.0.1 --port 8000
#
# In 1Claw, ensure the agent can read:
#   mercury/rpc/base (and/or mercury/rpc/ethereum)
#   mercury/wallets/<wallet_id>/private_key   (only needed for signing / value-moving flows)
# Optional (swap providers): mercury/apis/lifi  — if unset, LiFi uses the public API tier
#   (no x-lifi-api-key header; lower rate limits than with a key from https://jumper.exchange/lifi
#   or your LiFi dashboard).
#
# Replace placeholders below:
#   YOUR_WALLET_ADDRESS — must match the EVM address for private key at mercury/wallets/primary/private_key
#   YOUR_RECIPIENT — any valid 0x address you control for transfer tests
#   BASE — http://127.0.0.1:8000 (or your deployed host)

# -----------------------------------------------------------------------------
# 1) Mint a short-lived JWT from 1Claw (optional sanity check outside Mercury)
#
# Confirms MERCURY_ONECLAW_AGENT_ID + API key pair. Use the access_token from the
# JSON response as Bearer for direct vault API calls (step 2).
# -----------------------------------------------------------------------------
curl -sS -X POST "${MERCURY_ONECLAW_BASE_URL:-https://api.1claw.xyz}/v1/auth/agent-token" \
  -H "Content-Type: application/json" \
  -d "{\"agent_id\": \"${MERCURY_ONECLAW_AGENT_ID}\", \"api_key\": \"${MERCURY_ONECLAW_API_KEY}\"}"

# -----------------------------------------------------------------------------
# 2) Read one RPC secret directly from 1Claw (optional; same path Mercury uses)
#
# Verifies vault id, agent token scopes, and that mercury/rpc/base exists. A 403
# usually means the agent is not granted that vault or path — fix in 1Claw before
# testing Mercury invokes.
# Replace ACCESS_TOKEN with the JWT from step 1.
# -----------------------------------------------------------------------------
curl -sS "${MERCURY_ONECLAW_BASE_URL:-https://api.1claw.xyz}/v1/vaults/${MERCURY_ONECLAW_VAULT_ID}/secrets/mercury/rpc/base" \
  -H "Authorization: Bearer ACCESS_TOKEN"

# -----------------------------------------------------------------------------
# 3) Liveness — process is up (no 1Claw call)
# -----------------------------------------------------------------------------
curl -sS http://127.0.0.1:8000/healthz

# -----------------------------------------------------------------------------
# 4) Readiness — settings + chain registry OK (still no wallet or RPC fetch)
# -----------------------------------------------------------------------------
curl -sS http://127.0.0.1:8000/readyz

# -----------------------------------------------------------------------------
# 4b) LiFi /quote only — direct to https://li.quest (no Mercury, no 1Claw)
#
# Exercises the same HTTP API the LiFi adapter uses (GET /v1/quote). Public tier:
# no API key. Replace FROM_ADDRESS with a real 0x (e.g. a test wallet). Amounts
# are raw token units: 1 USDC on Base = 1000000 (6 decimals).
# Docs: https://docs.li.fi/
# -----------------------------------------------------------------------------
LIFI_WALLET="0x0000000000000000000000000000000000000001"
curl -sS -G "https://li.quest/v1/quote" \
  --data-urlencode "fromChain=8453" \
  --data-urlencode "toChain=8453" \
  --data-urlencode "fromToken=0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913" \
  --data-urlencode "toToken=0x4200000000000000000000000000000000000006" \
  --data-urlencode "fromAmount=1000000" \
  --data-urlencode "fromAddress=${LIFI_WALLET}" \
  --data-urlencode "toAddress=${LIFI_WALLET}" \
  --data-urlencode "slippage=0.01" \
  | (command -v jq >/dev/null && jq . || cat -)

# 10b) Same ERC20 transfer — submit human approval (second request)
#
# After (10), if the response is approval_required, repeat the SAME intent and the
# SAME Idempotency-Key as (10), and add a top-level approval_response. Mercury
# merges that into transaction metadata and the approver will only sign when
# status is "approved". Optional: set approval_response.idempotency_key to the same
# key as the Idempotency-Key header to bind approval to this attempt.
#
# Use a fixed key for both calls, e.g. export IDEM=e2e-erc20-transfer-manual-1
# -----------------------------------------------------------------------------

  -H "Content-Type: application/json" \

# 11) Swap quote / prepare — Mercury + LiFi (value-moving; needs 1Claw + RPC)
#
# Prerequisites: Mercury running (see header), 1Claw with mercury/rpc/base, wallet
# address matching mercury/wallets/primary/private_key if you go past “prepare”.
# Idempotency-Key (header) is merged into the swap intent; include idempotency_key
# in the body too for clarity. provider_preference "lifi" uses LiFi; bridge routes
# are rejected unless policy enables allow_bridges.

# Expect: JSON with prepared_swap (quote, allowance, next tx) or policy/approval
# state — not a raw LiFi payload (that is step 4b). For a fresh attempt, change
# both Idempotency-Key and intent.idempotency_key (e.g. e2e-swap-lifi-2). For an

# -----------------------------------------------------------------------------
# 11A) End-to-end LiFi swap (same-chain) — what runs and in what order
#
# With provider_preference "lifi", one HTTP invoke runs a LangGraph that:
#   1) Quotes via LiFi and checks ERC20 allowance to the provider spender.
#   2) If allowance is too low, the *next* transaction to sign is an ERC20 approval
#      (not the swap). Human approval, sign, and broadcast can complete in one
#      invoke (steps below) for that approval tx, or you run a second invoke after
#      the on-chain approval confirms (use a *new* idempotency key for a fresh swap).
#   3) If allowance is already enough, the prepared transaction is the LiFi swap;
#      the pipeline is: nonce → gas → simulation → policy → human approval
#      (RequestMetadataTransactionApprover in the default service) → idempotency
#      → sign (1Claw + wallet key) → broadcast → receipt.
#
# Human approval: the default app uses ``RequestMetadataTransactionApprover``. The
# first response may include ``approval_payload``; repeat the *same* swap intent
# and the *same* Idempotency-Key, and set top-level ``approval_response`` with
# status "approved" (see 11B). The approver also checks that
# ``approval_response.idempotency_key`` matches the transaction idempotency key when
# both are set.
#
# If you need *two* on-chain steps (approval tx, then swap), expect two pairs of
# calls: (request → approve+execute approval) with idem ``…-approve-1``,
# then (request → approve+execute swap) with idem ``…-swap-1``. Pre-approving the
# spender in the wallet (outside Mercury) lets you skip the first pair.
# -----------------------------------------------------------------------------

# -----------------------------------------------------------------------------
# 11B) E2E step 1 — request swap; expect approval gate or a prepared path in data
# -----------------------------------------------------------------------------
# export BASE=http://127.0.0.1:8000
# curl (same as 11) ...

# -----------------------------------------------------------------------------
# 11C) E2E step 2 — same intent and keys, add human approval so signing can proceed
#
# Run this only after 11 returned ``approval_required`` or equivalent. Reuse the
# exact ``Idempotency-Key`` and ``intent.idempotency_key`` from 11. Adjust keys if
# you use different ones in step 1.
# -----------------------------------------------------------------------------
curl -sS -X POST http://127.0.0.1:8000/v1/mercury/invoke \
  -H "Content-Type: application/json" \
  -H "X-Request-ID: e2e-swap-1b" \
  -H "Idempotency-Key: e2e-swap-lifi-1" \
  -d '{
    "user_id": "e2e-user",
    "wallet_id": "primary",
    "approval_response": {
      "status": "approved",
      "approved_by": "e2e-operator",
      "reason": "Out-of-band confirmation for swap or approval transaction",
      "idempotency_key": "e2e-swap-lifi-1"
    },
    "intent": {
      "kind": "swap",
      "chain": "base",
      "wallet_id": "primary",
      "from_token": "0x833589fCD6eDb6E08f4c7C32D4f71b54bdA02913",
      "to_token": "0x4200000000000000000000000000000000000006",
      "amount_in": "1",
      "max_slippage_bps": 100,
      "provider_preference": "lifi",
      "idempotency_key": "e2e-swap-lifi-1"
    }
  }'
